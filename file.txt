```java
package com.rushbooyah.vsappstudio;

import android.util.Base64;
import java.nio.charset.StandardCharsets;
import java.security.MessageDigest;
import java.security.SecureRandom;
import java.util.Arrays;
import javax.crypto.Cipher;
import javax.crypto.Mac;
import javax.crypto.spec.IvParameterSpec;
import javax.crypto.spec.SecretKeySpec;

public final class SecurityUtil {
    private SecurityUtil(){}

    public static String hmac(String data, String key){
        try{
            Mac mac = Mac.getInstance("HmacSHA256");
            SecretKeySpec secret = new SecretKeySpec(key.getBytes(StandardCharsets.UTF_8), "HmacSHA256");
            mac.init(secret);
            byte[] result = mac.doFinal(data.getBytes(StandardCharsets.UTF_8));
            StringBuilder sb = new StringBuilder();
            for(byte b : result) sb.append(String.format("%02x", b & 0xff));
            return sb.toString();
        }catch(Exception e){ return ""; }
    }

    public static String aesEncrypt(String plainText, String keyStr){
        try{
            byte[] keyBytes = sha256(keyStr.getBytes(StandardCharsets.UTF_8));
            SecretKeySpec key = new SecretKeySpec(keyBytes, "AES");
            byte[] iv = new byte[16];
            new SecureRandom().nextBytes(iv);
            Cipher cipher = Cipher.getInstance("AES/CBC/PKCS5Padding");
            cipher.init(Cipher.ENCRYPT_MODE, key, new IvParameterSpec(iv));
            byte[] encrypted = cipher.doFinal(plainText.getBytes(StandardCharsets.UTF_8));
            byte[] combined = new byte[iv.length + encrypted.length];
            System.arraycopy(iv, 0, combined, 0, iv.length);
            System.arraycopy(encrypted, 0, combined, iv.length, encrypted.length);
            return Base64.encodeToString(combined, Base64.NO_WRAP);
        }catch(Exception e){ return ""; }
    }

    public static String aesDecrypt(String encryptedBase64, String keyStr){
        try{
            byte[] keyBytes = sha256(keyStr.getBytes(StandardCharsets.UTF_8));
            SecretKeySpec key = new SecretKeySpec(keyBytes, "AES");
            byte[] combined = Base64.decode(encryptedBase64, Base64.NO_WRAP);
            byte[] iv = Arrays.copyOfRange(combined, 0, 16);
            byte[] encrypted = Arrays.copyOfRange(combined, 16, combined.length);
            Cipher cipher = Cipher.getInstance("AES/CBC/PKCS5Padding");
            cipher.init(Cipher.DECRYPT_MODE, key, new IvParameterSpec(iv));
            return new String(cipher.doFinal(encrypted), StandardCharsets.UTF_8);
        }catch(Exception e){ return ""; }
    }

    public static String createSignature(String encryptedPayload, String time, String nonce, String sessionKey){
        String raw = encryptedPayload + "|" + time + "|" + nonce + "|" + sessionKey;
        return hmac(raw, vscfg.APP_PUBLIC_KEY);
    }

    private static byte[] sha256(byte[] input){
        try{ return MessageDigest.getInstance("SHA-256").digest(input); }
        catch(Exception e){ return new byte[32]; }
    }
}
```

```java
package com.rushbooyah.vsappstudio;

import android.content.Context;
import android.net.ConnectivityManager;
import android.net.Network;
import android.net.NetworkCapabilities;

public final class NetworkUtil{
    private NetworkUtil(){}
    public static boolean online(Context c){
        try{
            ConnectivityManager cm=(ConnectivityManager)c.getSystemService(Context.CONNECTIVITY_SERVICE);
            Network n=cm.getActiveNetwork();
            if(n==null)return false;
            NetworkCapabilities nc=cm.getNetworkCapabilities(n);
            return nc!=null&&(nc.hasTransport(NetworkCapabilities.TRANSPORT_WIFI)||nc.hasTransport(NetworkCapabilities.TRANSPORT_CELLULAR));
        }catch(Exception e){ return false; }
    }
}
```

```java
package com.rushbooyah.vsappstudio;

import android.os.Debug;
import java.io.File;

public final class RootCheck {
    private RootCheck(){}
    public static boolean isRooted(){
        return checkSuBinary() || checkTestKeys() || checkFrida();
    }
    private static boolean checkSuBinary(){
        String[] paths = {"/system/bin/su","/system/xbin/su","/sbin/su","/su/bin/su","/data/local/xbin/su","/data/local/bin/su"};
        for(String p:paths) if(new File(p).exists()) return true;
        return false;
    }
    private static boolean checkTestKeys(){
        return android.os.Build.TAGS != null && android.os.Build.TAGS.contains("test-keys");
    }
    private static boolean checkFrida(){
        return Debug.isDebuggerConnected();
    }
}
```

```java
package com.rushbooyah.vsappstudio;

import android.content.Context;
import android.content.SharedPreferences;

public class SessionManager {
    private final SharedPreferences p;
    private static final long SESSION_DURATION = 24 * 60 * 60 * 1000;

    public SessionManager(Context c){
        p = c.getSharedPreferences("VS_SESSION", Context.MODE_PRIVATE);
    }

    public void save(String uid, String loginKey, String sessionKey, String name,
                     String email, String machineId, long balance, long loginTime){
        p.edit()
            .putString("uid", uid)
            .putString("loginKey", loginKey)
            .putString("sessionKey", sessionKey)
            .putString("name", name)
            .putString("email", email)
            .putString("machineId", machineId)
            .putLong("balance", balance)
            .putLong("loginTime", loginTime)
            .commit();
    }

    public void updateBalance(long balance){
        p.edit().putLong("balance", balance).apply();
    }

    public String uid(){ return p.getString("uid",""); }
    public String loginKey(){ return p.getString("loginKey",""); }
    public String sessionKey(){ return p.getString("sessionKey",""); }
    public String name(){ return p.getString("name",""); }
    public String email(){ return p.getString("email",""); }
    public String machineId(){ return p.getString("machineId",""); }
    public long balance(){ return p.getLong("balance",0); }
    public long loginTime(){ return p.getLong("loginTime",0); }

    public boolean isLoggedIn(){
        return !uid().isEmpty() && !loginKey().isEmpty() && !sessionKey().isEmpty();
    }

    public boolean isSessionExpired(){
        long lt = loginTime();
        return lt == 0 || (System.currentTimeMillis() - lt) > SESSION_DURATION;
    }

    public void clearSession(){ p.edit().clear().commit(); }
    public void logout(){ clearSession(); }
}
```

`ApiClient.java`

```java
package com.rushbooyah.vsappstudio;

import android.content.Context;
import org.json.JSONObject;
import java.io.IOException;
import java.util.UUID;
import java.util.concurrent.TimeUnit;
import okhttp3.Call;
import okhttp3.Callback;
import okhttp3.MediaType;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.RequestBody;
import okhttp3.Response;

public final class ApiClient {

    private static final OkHttpClient client = new OkHttpClient.Builder()
        .connectTimeout(vscfg.TIMEOUT, TimeUnit.MILLISECONDS)
        .readTimeout(vscfg.TIMEOUT, TimeUnit.MILLISECONDS)
        .writeTimeout(vscfg.TIMEOUT, TimeUnit.MILLISECONDS)
        .build();

    public static final MediaType JSON = MediaType.parse("application/json; charset=utf-8");

    public interface ApiCallback{
        void success(JSONObject data);
        void error(String msg);
    }

    private ApiClient(){}

    public static void post(Context c, String url, JSONObject body, ApiCallback cb){
        if(!NetworkUtil.online(c)){ cb.error("NETWORK_OFF"); return; }
        if(RootCheck.isRooted()){ cb.error("SECURITY_BREACH"); return; }

        SessionManager sm = new SessionManager(c);
        if(!sm.isLoggedIn() || sm.isSessionExpired()){
            cb.error("SESSION_EXPIRED");
            return;
        }

        String uid = sm.uid();
        String loginKey = sm.loginKey();
        String sessionKey = sm.sessionKey();
        String machineId = sm.machineId();
        long loginTime = sm.loginTime();

        String time = aabb.time();
        String nonce = aabb.nonce();
        String requestId = UUID.randomUUID().toString().replace("-","").substring(0,12);

        String payloadEncrypted = SecurityUtil.aesEncrypt(body.toString(), loginKey);
        String authPacket = uid + "|" + loginKey + "|" + sessionKey + "|" + machineId + "|" + loginTime;
        String authEncrypted = SecurityUtil.aesEncrypt(authPacket, vscfg.APP_PUBLIC_KEY);
        String signature = SecurityUtil.createSignature(payloadEncrypted, time, nonce, sessionKey);

        JSONObject wrapper = new JSONObject();
        try{ wrapper.put("_", payloadEncrypted); }catch(Exception e){}

        Request request = new Request.Builder()
            .url(url)
            .addHeader("appKey", vscfg.APP_PUBLIC_KEY)
            .addHeader("x-auth", authEncrypted)
            .addHeader("time", time)
            .addHeader("nonce", nonce)
            .addHeader("signature", signature)
            .addHeader("x-request-id", requestId)
            .post(RequestBody.create(wrapper.toString(), JSON))
            .build();

        enqueue(request, sm, cb);
    }

    public static void postPublic(Context c, String url, JSONObject body, ApiCallback cb){
        if(!NetworkUtil.online(c)){ cb.error("NETWORK_OFF"); return; }
        if(RootCheck.isRooted()){ cb.error("SECURITY_BREACH"); return; }

        String time = aabb.time();
        String nonce = aabb.nonce();
        String requestId = UUID.randomUUID().toString().replace("-","").substring(0,12);

        String payloadEncrypted = SecurityUtil.aesEncrypt(body.toString(), vscfg.APP_PUBLIC_KEY);
        String signature = SecurityUtil.createSignature(payloadEncrypted, time, nonce, vscfg.APP_PUBLIC_KEY);

        JSONObject wrapper = new JSONObject();
        try{ wrapper.put("_", payloadEncrypted); }catch(Exception e){}

        Request request = new Request.Builder()
            .url(url)
            .addHeader("appKey", vscfg.APP_PUBLIC_KEY)
            .addHeader("time", time)
            .addHeader("nonce", nonce)
            .addHeader("signature", signature)
            .addHeader("x-request-id", requestId)
            .post(RequestBody.create(wrapper.toString(), JSON))
            .build();

        enqueue(request, null, cb);
    }

    private static void enqueue(Request request, SessionManager sm, ApiCallback cb){
        client.newCall(request).enqueue(new Callback(){
            @Override public void onFailure(Call call, IOException e){ cb.error("NETWORK_ERROR"); }
            @Override public void onResponse(Call call, Response response) throws IOException {
                String r = response.body() != null ? response.body().string() : "";
                try{
                    JSONObject resp = new JSONObject(r);
                    if(resp.has("payload") && sm != null && sm.isLoggedIn()){
                        String decrypted = SecurityUtil.aesDecrypt(resp.optString("payload"), sm.loginKey());
                        if(decrypted.isEmpty()){ cb.error("DECRYPT_ERROR"); return; }
                        cb.success(new JSONObject(decrypted));
                    } else {
                        cb.success(resp);
                    }
                }catch(Exception e){ cb.error("RESPONSE_ERROR"); }
            }
        });
    }
}
```

```java
package com.rushbooyah.vsappstudio;

import android.app.Activity;
import android.content.Intent;
import android.os.Bundle;
import android.provider.Settings;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;
import android.widget.Toast;

import org.json.JSONObject;

public class LoginActivity extends Activity {

    EditText nameInput, emailInput, passwordInput;
    Button actionBtn;
    TextView signupText, verifyText, promptText, titleText;
    SessionManager sm;
    String machineId;
    boolean isSignupMode = false;
    String lastEmail = "", lastPassword = "";

    @Override
    protected void onCreate(Bundle b){
        super.onCreate(b);
        setContentView(R.layout.login);

        sm = new SessionManager(this);
        if(sm.isLoggedIn()){ goHome(); return; }

        machineId = Settings.Secure.getString(getContentResolver(), Settings.Secure.ANDROID_ID);

        nameInput = findViewById(R.id.name);
        emailInput = findViewById(R.id.email);
        passwordInput = findViewById(R.id.password);
        actionBtn = findViewById(R.id.login);
        signupText = findViewById(R.id.signup);
        verifyText = findViewById(R.id.verify_email);
        promptText = findViewById(R.id.textview5);
        titleText = findViewById(R.id.textview4);

        verifyText.setVisibility(View.GONE);

        actionBtn.setOnClickListener(v -> {
            if(isSignupMode) signup();
            else login();
        });

        signupText.setOnClickListener(v -> toggleMode());
        verifyText.setOnClickListener(v -> resendVerification());
    }

    private void toggleMode(){
        isSignupMode = !isSignupMode;
        if(isSignupMode){
            nameInput.setVisibility(View.VISIBLE);
            actionBtn.setText("CREATE ACCOUNT");
            signupText.setText("Login");
            promptText.setText("Already have an account? ");
            titleText.setText("Sign Up");
            verifyText.setVisibility(View.GONE);
        } else {
            nameInput.setVisibility(View.GONE);
            actionBtn.setText("LOGIN");
            signupText.setText("Sign Up");
            promptText.setText("Don't have an account? ");
            titleText.setText("Login");
            verifyText.setVisibility(View.GONE);
        }
    }

    private void signup(){
        try{
            String name = nameInput.getText().toString().trim();
            String email = emailInput.getText().toString().trim();
            String pass = passwordInput.getText().toString().trim();

            if(name.isEmpty() || email.isEmpty() || pass.isEmpty()){
                Toast.makeText(this, "Fill all fields", Toast.LENGTH_SHORT).show();
                return;
            }

            JSONObject data = new JSONObject();
            data.put("name", name);
            data.put("email", email);
            data.put("password", pass);
            data.put("machineId", machineId);

            ApiClient.postPublic(this, Endpoint.signup(), data, new ApiClient.ApiCallback(){
                @Override public void success(JSONObject r){
                    runOnUiThread(() -> {
                        Toast.makeText(LoginActivity.this, r.optString("message"), Toast.LENGTH_LONG).show();
                        if(r.optBoolean("status")) toggleMode();
                    });
                }
                @Override public void error(String e){
                    runOnUiThread(() -> Toast.makeText(LoginActivity.this, e, Toast.LENGTH_SHORT).show());
                }
            });
        }catch(Exception e){
            Toast.makeText(this, "ERROR", Toast.LENGTH_SHORT).show();
        }
    }

    private void login(){
        try{
            String email = emailInput.getText().toString().trim();
            String pass = passwordInput.getText().toString().trim();

            if(email.isEmpty() || pass.isEmpty()){
                Toast.makeText(this, "Enter email and password", Toast.LENGTH_SHORT).show();
                return;
            }

            lastEmail = email;
            lastPassword = pass;

            JSONObject data = new JSONObject();
            data.put("email", email);
            data.put("password", pass);
            data.put("machineId", machineId);

            ApiClient.postPublic(this, Endpoint.login(), data, new ApiClient.ApiCallback(){
                @Override public void success(JSONObject r){
                    runOnUiThread(() -> {
                        if(r.optBoolean("status")){
                            sm.clearSession();
                            sm.save(
                                r.optString("uid"),
                                r.optString("loginKey"),
                                r.optString("sessionKey"),
                                r.optString("name"),
                                r.optString("email"),
                                machineId,
                                r.optLong("balance"),
                                r.optLong("loginTime")
                            );
                            goHome();
                        } else {
                            String msg = r.optString("message");
                            Toast.makeText(LoginActivity.this, msg, Toast.LENGTH_LONG).show();
                            if("EMAIL_NOT_VERIFIED".equals(msg)) verifyText.setVisibility(View.VISIBLE);
                        }
                    });
                }
                @Override public void error(String e){
                    runOnUiThread(() -> Toast.makeText(LoginActivity.this, e, Toast.LENGTH_SHORT).show());
                }
            });
        }catch(Exception e){
            Toast.makeText(this, "LOGIN ERROR", Toast.LENGTH_SHORT).show();
        }
    }

    private void resendVerification(){
        if(lastEmail.isEmpty() || lastPassword.isEmpty()){
            Toast.makeText(this, "Enter email & password first", Toast.LENGTH_SHORT).show();
            return;
        }
        try{
            JSONObject data = new JSONObject();
            data.put("email", lastEmail);
            data.put("password", lastPassword);

            ApiClient.postPublic(this, Endpoint.resendVerify(), data, new ApiClient.ApiCallback(){
                @Override public void success(JSONObject r){
                    runOnUiThread(() -> Toast.makeText(LoginActivity.this, r.optString("message"), Toast.LENGTH_LONG).show());
                }
                @Override public void error(String e){
                    runOnUiThread(() -> Toast.makeText(LoginActivity.this, e, Toast.LENGTH_SHORT).show());
                }
            });
        }catch(Exception e){
            Toast.makeText(this, "Error", Toast.LENGTH_SHORT).show();
        }
    }

    private void goHome(){
        startActivity(new Intent(this, HomeActivity.class));
        finish();
    }
}
```

```java
                runOnUiThread(() -> {
                    if (!r.optBoolean("status")) {
                        Toast.makeText(HomeActivity.this, r.optString("message"), Toast.LENGTH_SHORT).show();
                        return;
                    }

                    JSONObject server = r.optJSONObject("server");
                    if (server != null) {
                        if (server.optBoolean("maintenance")) {
                            BlockDialog.show(HomeActivity.this, server.optString("notice", "Server Under Maintenance"));
                            return;
                        }
                        String latest = server.optString("latestVersion", vscfg.VERSION);
                        if (!vscfg.VERSION.equals(latest)) {
                            BlockDialog.show(HomeActivity.this, "Please update the app to the latest version.");
                            return;
                        }
                    }

                    JSONObject profile = r.optJSONObject("profile");
                    if (profile != null) {
                        if (!profile.optBoolean("verified", false)) {
                            Toast.makeText(HomeActivity.this, "Please verify your email", Toast.LENGTH_LONG).show();
                            sm.logout();
                            goLogin();
                            return;
                        }
                        nameText.setText(profile.optString("name"));
                        emailText.setText(profile.optString("email"));
                    }

                    long balance = r.optLong("balance");
                    balanceText.setText("₹ " + balance);
                    sm.updateBalance(balance);

                    tournaments.clear();
                    JSONObject obj = r.optJSONObject("tournaments");
                    if (obj != null) {
                        Iterator<String> keys = obj.keys();
                        while (keys.hasNext()) {
                            String id = keys.next();
                            JSONObject t = obj.optJSONObject(id);
                            if (t != null) {
                                try { t.put("id", id); } catch (Exception e) {}
                                tournaments.add(t);
                            }
                        }
                    }
                    adapter.notifyDataSetChanged();
                });
            }

            @Override
            public void error(String e) {
                runOnUiThread(() -> {
                    if ("NETWORK_OFF".equals(e)) {
                        BlockDialog.network(HomeActivity.this);
                    } else {
                        Toast.makeText(HomeActivity.this, e, Toast.LENGTH_SHORT).show();
                    }
                });
            }
        });
    }
```

`utils/crypto.js`

```javascript
const crypto = require("crypto");

function sha256(input) {
    return crypto.createHash("sha256").update(input).digest();
}

function aesEncrypt(text, keyStr) {
    const key = sha256(keyStr);
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipheriv("aes-256-cbc", key, iv);
    let encrypted = cipher.update(text, "utf8", "base64");
    encrypted += cipher.final("base64");
    const combined = Buffer.concat([iv, Buffer.from(encrypted, "base64")]);
    return combined.toString("base64");
}

function aesDecrypt(encryptedBase64, keyStr) {
    const key = sha256(keyStr);
    const combined = Buffer.from(encryptedBase64, "base64");
    const iv = combined.slice(0, 16);
    const encrypted = combined.slice(16);
    const decipher = crypto.createDecipheriv("aes-256-cbc", key, iv);
    let decrypted = decipher.update(encrypted, "base64", "utf8");
    decrypted += decipher.final("utf8");
    return decrypted;
}

function hmac(data, key) {
    return crypto.createHmac("sha256", key).update(data, "utf8").digest("hex");
}

module.exports = { aesEncrypt, aesDecrypt, hmac };
```

```javascript
async function updateReward(tournamentId) {
    const ref = db.ref("tournaments/" + tournamentId);
    const snap = await ref.once("value");
    if (!snap.exists()) return;

    const t = snap.val();
    const minPlayers = Number(t.minPlayers || 0);
    const maxPlayers = Number(t.maxPlayers || 0);
    const joinedPlayers = Number(t.joinedPlayers || 0);
    const entryFee = Number(t.entryFee || 0);
    const platformFee = Number(t.platformFee || 20);
    const perKillPercent = Number(t.perKillPercent || 20);

    if (joinedPlayers < minPlayers) {
        await ref.child("reward").set({
            active: false,
            waiting: true,
            remaining: minPlayers - joinedPlayers,
            joinedPlayers,
            minPlayers,
            prizePool: 0,
            perKill: 0,
            rank: {},
            updatedAt: Date.now()
        });
        return;
    }

    const totalCollection = joinedPlayers * entryFee;
    const adminAmount = Math.floor(totalCollection * platformFee / 100);
    const prizePool = totalCollection - adminAmount;
    const killPool = Math.floor(prizePool * perKillPercent / 100);
    const rankPool = prizePool - killPool;

    const rank = {};
    rank[1] = Math.floor(rankPool * 0.50);
    rank[2] = Math.floor(rankPool * 0.25);
    rank[3] = Math.floor(rankPool * 0.15);

    let remain = rankPool - rank[1] - rank[2] - rank[3];
    const extra = Math.min(7, Math.max(joinedPlayers - 3, 0));

    for (let i = 4; i <= 3 + extra; i++) {
        rank[i] = Math.floor(remain / extra);
    }

    await ref.child("reward").set({
        active: true,
        waiting: false,
        joinedPlayers,
        minPlayers,
        maxPlayers,
        prizePool,
        adminAmount,
        perKill: killPool,
        rank,
        updatedAt: Date.now()
    });
}
```

```javascript
const config = require("../config");
const { checkNonce } = require("../utils/nonce");

async function security(req, res, next) {
    try {
        const appKey = req.headers["appkey"] || req.headers["app-key"];
        const time = req.headers["time"];
        const nonce = req.headers["nonce"];

        if (appKey !== config.SECURITY.APP_PUBLIC_KEY) {
            return res.status(401).json({ status: false, message: "APP_BLOCKED" });
        }
        if (!time || !nonce) {
            return res.status(401).json({ status: false, message: "SECURITY_DATA_MISSING" });
        }

        const requestTime = Number(time);
        if (Number.isNaN(requestTime)) {
            return res.status(401).json({ status: false, message: "INVALID_TIME" });
        }

        const diff = Date.now() - requestTime;
        if (diff < 0 || diff > 10000) {
            return res.status(401).json({ status: false, message: "REQUEST_EXPIRED" });
        }

        const validNonce = await checkNonce(nonce);
        if (!validNonce) {
            return res.status(401).json({ status: false, message: "REPLAY_BLOCKED" });
        }

        next();
    } catch (e) {
        return res.status(401).json({ status: false, message: "SECURITY_FAILED" });
    }
}
```

`middleware/auth.js`

```javascript
const { aesDecrypt, hmac } = require("../utils/crypto");
const { getSession } = require("../utils/session");
const config = require("../config");

async function auth(req, res, next) {
    try {
        const xAuth = req.headers["x-auth"];
        const time = req.headers["time"];
        const nonce = req.headers["nonce"];
        const signature = req.headers["signature"];

        if (!xAuth || !signature) {
            return res.status(401).json({ status: false, message: "AUTH_DATA_MISSING" });
        }

        let authPacket;
        try {
            authPacket = aesDecrypt(xAuth, config.SECURITY.APP_PUBLIC_KEY);
        } catch (e) {
            return res.status(401).json({ status: false, message: "INVALID_AUTH_PACKET" });
        }

        const parts = authPacket.split("|");
        if (parts.length !== 5) {
            return res.status(401).json({ status: false, message: "INVALID_AUTH_PACKET" });
        }

        const [uid, loginKey, sessionKey, machineId, loginTime] = parts;

        const session = await getSession(uid);
        if (!session) {
            return res.status(401).json({ status: false, message: "SESSION_NOT_FOUND" });
        }

        if (session.loginKey !== loginKey || session.sessionKey !== sessionKey || session.machineId !== machineId) {
            return res.status(401).json({ status: false, message: "DEVICE_BLOCKED" });
        }

        const encryptedPayload = req.body._;
        if (!encryptedPayload) {
            return res.status(400).json({ status: false, message: "PAYLOAD_MISSING" });
        }

        const expectedSig = hmac(encryptedPayload + "|" + time + "|" + nonce + "|" + sessionKey, config.SECURITY.APP_PUBLIC_KEY);
        if (expectedSig !== signature) {
            return res.status(401).json({ status: false, message: "INVALID_SIGNATURE" });
        }

        let decryptedBody;
        try {
            decryptedBody = aesDecrypt(encryptedPayload, loginKey);
        } catch (e) {
            return res.status(400).json({ status: false, message: "PAYLOAD_DECRYPT_FAILED" });
        }

        try {
            req.body = JSON.parse(decryptedBody);
        } catch (e) {
            return res.status(400).json({ status: false, message: "INVALID_JSON" });
        }

        req.user = { uid, loginKey, sessionKey, machineId, loginTime: Number(loginTime) };
        next();

    } catch (e) {
        console.log("AUTH ERROR:", e.message);
        return res.status(401).json({ status: false, message: "AUTH_FAILED" });
    }
}

module.exports = auth;
```

```javascript
        if (!/^[a-zA-Z0-9._-]+@[a-zA-Z]+$/.test(upi)) {
            return fail(res, "Invalid UPI ID format");
        }

        const balSnap = await db.ref("bal/" + uid).once("value");
        const currentBal = balSnap.val() || { balance: 0 };
        const balance = Number(currentBal.balance || 0);

        if (balance < value) return fail(res, "Insufficient Balance");

        const newBalance = balance - value;
        const id = "WD_" + Date.now();

        const updates = {};
        updates["bal/" + uid + "/balance"] = newBalance;
        updates["withdraw_requests/" + id] = {
            uid, amount: value, upi,
            status: "pending",
            createdAt: Date.now()
        };

        await db.ref().update(updates);

        const responseData = {
            status: true,
            message: "Withdraw Request Submitted",
            withdrawId: id
        };

        const encryptedPayload = aesEncrypt(JSON.stringify(responseData), req.user.loginKey);
        return res.json({ status: true, payload: encryptedPayload });

    } catch (e) { return fail(res, "Withdraw Failed"); }
});

module.exports = router;
```
